# [[백준 2309] 일곱 난쟁이](https://www.acmicpc.net/problem/2309)
## JavaScript
```js
const arr = require('fs').readFileSync('./stdin').toString().trim().split(/\s+/).map(v => +v);

const solve = arr => {

  // 아홉 난쟁이의 키의 합을 구함
  const sum = arr.reduce((acc, v) => acc + v, 0);

  // 레이븖 문(이중 for문을 탈출하기 위해서)
  loop:
  for (let i = 0; i < 8; i++) {
    for (let j = i + 1; j < 9; j++) {

      // 서로 다른 두 난쟁이의 키의 합을 sum에서 뺀 값이 100이면
      if (sum - arr[i] - arr[j] === 100) {

        // arr[i], arr[j]를 제외한 난쟁이들의 키를 오름차순 정렬하고 개행문자로 구분하여 출력
        console.log(
          arr
            .filter(v => v !== arr[i] && v !== arr[j])
            .sort((a, b) => a - b)
            .join('\n')
        );

        // 이중 for문 탈출
        break loop;
      } 
    }
  }
};

solve(arr);
```
## Python
```py
arr = [int(input()) for _ in range(9)]

def solve(arr):
    for i in range(len(arr) - 1):
        for j in range(i + 1, len(arr)):
            if sum(arr) - arr[i] - arr[j] == 100:
                arr.remove(arr[j])
                arr.remove(arr[i])
                print('\n'.join(map(str, sorted(arr))))
                break

        # 이중 for문을 탈출하기 위해서 for-else문 사용
        else:
            continue
        break
    
solve(arr)
```
