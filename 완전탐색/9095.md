## Python
```py
import sys

T = int(sys.stdin.readline())
n = [int(input()) for _ in range(T)]

def solve(T, n):
    dp = [0 for _ in range(max(n)+1)]
    dp[1], dp[2], dp[3] = 1, 2, 4

    for i in range(4, max(n)+1):
        dp[i] = dp[i-1] + dp[i-2] + dp[i-3]

    print('\n'.join(map(lambda x: str(dp[x]), n)))

solve(T, n)
```
